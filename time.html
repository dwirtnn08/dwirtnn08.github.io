Data Prepocessing

Tipe data Date yang sebelumnya tipe datanya chr/ character diubah dengan menggunakan fungsi mdy_hms() dengan library lubridate.

# ubah datetime 
microsoft$DateC <- mdy_hms(microsoft$Date)
microsoft$DateClean <- format(as.Date(microsoft$DateC), format = "%Y-%m-%d")
microsoft <- microsoft %>% 
  select(-c(Date, DateC, Open, Low, Close, Volume)) %>% 
  mutate(DateClean = lubridate::ymd(DateClean)) %>% 
  arrange(DateClean) %>% 
  padr::pad()
